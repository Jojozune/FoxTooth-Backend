Installation & Setup

1

Create Backend Manager Script

Create a new C# script called GameBackendManager.cs in your Assets folder:

GameBackendManager.cs
using UnityEngine; using UnityEngine.Networking; using System.Collections; using System.Collections.Generic; public class GameBackendManager : MonoBehaviour { public static GameBackendManager Instance { get; private set; } private const string API_BASE_URL = "http://localhost:7777"; private string accessToken = ""; private string refreshToken = ""; private string rememberToken = ""; private int playerId = 0; private void Awake() { if (Instance != null && Instance != this) { Destroy(gameObject); return; } Instance = this; DontDestroyOnLoad(gameObject); } private void Start() { LoadSavedTokens(); } private void SaveTokens() { PlayerPrefs.SetString("access_token", accessToken); PlayerPrefs.SetString("refresh_token", refreshToken); PlayerPrefs.SetString("remember_token", rememberToken); PlayerPrefs.SetInt("player_id", playerId); PlayerPrefs.Save(); } private void LoadSavedTokens() { accessToken = PlayerPrefs.GetString("access_token", ""); refreshToken = PlayerPrefs.GetString("refresh_token", ""); rememberToken = PlayerPrefs.GetString("remember_token", ""); playerId = PlayerPrefs.GetInt("player_id", 0); } private void ClearTokens() { accessToken = ""; refreshToken = ""; rememberToken = ""; playerId = 0; PlayerPrefs.DeleteKey("access_token"); PlayerPrefs.DeleteKey("refresh_token"); PlayerPrefs.DeleteKey("remember_token"); PlayerPrefs.DeleteKey("player_id"); PlayerPrefs.Save(); } }

Account Creation

Add this method to create new player accounts:

Create Account Method
[System.Serializable] public class CreateAccountRequest { public string display_name; public string player_tag; public string email; public string password; } public IEnumerator CreateAccount( string displayName, string playerTag, string email, string password) { var request = new CreateAccountRequest { display_name = displayName, player_tag = playerTag, email = email, password = password }; string json = JsonUtility.ToJson(request); using (UnityWebRequest www = new UnityWebRequest( $"{API_BASE_URL}/account/create", "POST")) { byte[] bodyRaw = System.Text.Encoding.UTF8.GetBytes(json); www.uploadHandler = new UploadHandlerRaw(bodyRaw); www.downloadHandler = new DownloadHandlerBuffer(); www.SetRequestHeader("Content-Type", "application/json"); yield return www.SendWebRequest(); if (www.result == UnityWebRequest.Result.Success) { var response = JsonUtility.FromJson<CreateAccountResponse>( www.downloadHandler.text); Debug.Log($"Account created! Player ID: {response.player_id}"); } else { Debug.LogError($"Account creation failed: {www.error}"); } } }

Heartbeat & Session Management

7

Keep Player Alive with Heartbeats

Implement heartbeat system to keep player connection active:

Heartbeat Manager
private Coroutine heartbeatCoroutine; private float heartbeatInterval = 30f; // Send every 30 seconds private void OnEnable() { if (heartbeatCoroutine == null) { heartbeatCoroutine = StartCoroutine(HeartbeatLoop()); } } private void OnDisable() { if (heartbeatCoroutine != null) { StopCoroutine(heartbeatCoroutine); heartbeatCoroutine = null; } } private IEnumerator HeartbeatLoop() { while (true) { yield return new WaitForSeconds(heartbeatInterval); StartCoroutine(SendHeartbeat()); } } private IEnumerator SendHeartbeat() { using (UnityWebRequest www = new UnityWebRequest( $"{API_BASE_URL}/player/heartbeat", "POST")) { www.uploadHandler = new UploadHandlerRaw(new byte[0]); www.downloadHandler = new DownloadHandlerBuffer(); www.SetRequestHeader("Content-Type", "application/json"); www.SetRequestHeader("Authorization", $"Bearer {accessToken}"); yield return www.SendWebRequest(); if (www.result != UnityWebRequest.Result.Success) { Debug.LogWarning("Heartbeat failed - player may be disconnected"); // Attempt token refresh StartCoroutine(RefreshToken()); } } }

Best Practices for Unity Integration

üîê Security

  • Never store tokens in PlayerPrefs - use SecurePrefs or encrypted storage
  • Use HTTPS only in production builds
  • Validate all user input before sending to server
  • Handle token expiration gracefully with refresh logic
  • Clear tokens on logout or when game exits

‚ö° Performance

  • Use object pooling for frequently created coroutines
  • Batch API calls when possible to reduce requests
  • Implement request queuing for rate limit protection
  • Cache player data locally and sync periodically
  • Use coroutines efficiently - don't create unnecessary waiting patterns

üéÆ UX Improvements

  • Show loading indicators during API calls
  • Implement offline mode with local caching
  • Auto-refresh tokens before they expire
  • Display friendly error messages to players
  • Implement exponential backoff for failed requests

üß™ Testing

  • Create unit tests for token management
  • Test network failures and timeouts
  • Verify heartbeat timing and reconnection logic
  • Load test with multiple concurrent players
  • Test on actual mobile devices with poor connectivity

Common Issues & Solutions

Issue Cause Solution
401 Unauthorized Token expired or invalid Refresh token automatically, redirect to login if refresh fails
429 Too Many Requests Rate limit exceeded Implement exponential backoff, respect rate limit headers
Player marked offline immediately No heartbeats being sent Enable heartbeat coroutine on Start(), check interval is 30 seconds
Can't receive invites Player not online or heartbeat expired Verify heartbeats are being sent, check player online status
Session code invalid Session expired (2 min timeout) Create new session or extend expiration on backend
